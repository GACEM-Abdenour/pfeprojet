\chapter*{Liste des abréviations et sigles}
\addcontentsline{toc}{chapter}{Liste des abréviations et sigles}

\begin{description}
    \item[BI] \textit{Business Intelligence} — intelligence décisionnelle.
    \item[DCSI] Direction centrale des systèmes d'information (Naftal).
    \item[ERD] \textit{Entity-Relationship Diagram} — diagramme entité-association.
    \item[ERP] \textit{Enterprise Resource Planning} — progiciel de gestion intégré.
    \item[FastCGI] Interface d'exécution des scripts (ex.\ PHP) avec le serveur web IIS.
    \item[GIA] Gestion des incidents applicatifs (plateforme objet de ce mémoire).
    \item[GPL] Gaz de pétrole liquéfié.
    \item[IIS] \textit{Internet Information Services} — rôle serveur web Microsoft.
    \item[ITIL] \textit{Information Technology Infrastructure Library} — référentiel de bonnes pratiques pour le service des SI.
    \item[ITSM] \textit{IT Service Management} — management des services informatiques.
    \item[KPI] \textit{Key Performance Indicator} — indicateur clé de performance.
    \item[MEF] Modèle d'évaluation fonctionnelle (Autissier \& Delaye) — diagnostic des fonctions support.
    \item[MOA] Maîtrise d'ouvrage.
    \item[MOE / MŒ] Maîtrise d'œuvre.
    \item[MTTR] \textit{Mean Time To Resolve} — temps moyen de résolution.
    \item[PDO] \textit{PHP Data Objects} — couche d'accès aux bases de données en PHP.
    \item[SI] Système d'information.
    \item[SIAD] Système d'information d'aide à la décision.
    \item[SPA] Société par actions.
    \item[SQL] \textit{Structured Query Language}.
    \item[UML] \textit{Unified Modeling Language}.
    \item[UX/UI] Expérience utilisateur / interface utilisateur.
\end{description}
